// --- ADMIN ACTIONS CU LOGICĂ DUALĂ DE EMAIL --- async function adminReqAction(rid, approved) { const s = await db.ref('pending_donations/' + rid).once('value'); const r = s.val(); if(!r) return; if(approved) { let giftCodeToMail = "CREDIT PE CONT"; let chosenTemplate = "template_750es5r"; // Default: Template Credite if(r.type === 'giftcard') { chosenTemplate = "template_ou7sc0j"; // Switch la Template Gift Card // Generăm cod Gift Card unic giftCodeToMail = "SLIDE-" + Math.random().toString(36).substr(2, 9).toUpperCase(); await db.ref('active_giftcards/' + giftCodeToMail).set({ amount: r.amount, timestamp: Date.now() }); } else { // Adăugare credite direct în baza de date const us = await db.ref('users/' + r.uid).once('value'); const uData = us.val() || { credits: 0, totalDonated: 0 }; await db.ref('users/' + r.uid).update({ credits: (uData.credits || 0) + r.amount, totalDonated: (uData.totalDonated || 0) + r.amount }); } // Parametrii pentru email const emailParams = { to_email: r.email, to_name: r.username, gift_code: giftCodeToMail, amount: r.amount, approved_by: user.username }; // Trimitere folosind Service ID-ul tău: service_2gd65tf emailjs.send("service_2gd65tf", chosenTemplate, emailParams) .then(() => { console.log("Email trimis cu succes către " + r.email); }) .catch(err => { console.error("Eroare EmailJS:", err); notify("EROARE LA TRIMITERE EMAIL!"); }); // Notificare pe Discord (Webhook Aprobări) fetch(WH_APROBARI, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ content: `✅ Plata de **${r.amount} Cr** pt <@${r.uid}> a fost aprobată de **${user.username}**! (${r.type.toUpperCase()})` }) }); notify("CERERE APROBATĂ! EMAIL TRIMIS."); } else { notify("CERERE RESPINSĂ!"); } // Ștergem cererea din lista de așteptare await db.ref('pending_donations/' + rid).remove(); }
